from unittest import TestCase 

from mock import patch, call 

from logging import getLogger 

 

from pygame2 import display 

 

logger = getLogger() 

 

 

class DisplayTests(TestCase): 

 

    def setUp(self): 

        display.PygameDisplay._instance = None 

 

    def test_init(self): 

        """init is a noop""" 

        self.assertIsNone(display.init()) 

 

    @patch("pygame2.display.pyglet") 

    def test_set_mode_initializes_display_singleton(self, mock_pyglet): 

        """set_mode initializes the display singleton and pyglet main window""" 

        res = (640, 480) 

        surface = display.set_mode(res) 

        self.assertIs(display.PygameDisplay._instance, surface) 

        self.assertEqual(display.PygameDisplay._instance.get_resolution(), res) 

 

        # make sure there can only be one instance of the display 

        one = display.PygameDisplay(None) 

        two = display.PygameDisplay(None) 

        three = display.PygameDisplay(None) 

        self.assertIs(one, two) 

        self.assertIs(two, three) 

        self.assertIs(one, three) 

        self.assertTrue(mock_pyglet.window.Window.called) 

        self.assertIs(display.PygameDisplay(None)._instance.window, 

                      mock_pyglet.window.Window.return_value) 

 

    @patch("pygame2.display.pyglet") 

    def test_set_caption(self, mock_pyglet): 

        """can set the pyglet window caption on the 

        singleton window instance""" 

        display.set_mode((0, 0)) 

        title = "Zort: The Explorer" 

        display.set_caption(title, icontitle="foobar") 

        window_instance = mock_pyglet.window.Window.return_value 

        window_instance.set_caption.assert_called_once_with(title) 

 

    @patch("pygame2.display.pyglet") 

    def test_get_caption(self, mock_pyglet): 

        """can get the caption from the pyglet window""" 

        display.set_mode((0, 0)) 

        mock_pyglet.window.Window.return_value.caption = "foobar" 

        self.assertEqual(display.get_caption(), "foobar") 

 

    @patch("pygame2.display.pyglet") 

    def test_toggle_fullscreen(self, mock_pyglet): 

        """can toggle the fullscreen state by repeatedly 

        calling toggle_fullscreen""" 

        screen = display.set_mode((0, 0)) 

        self.assertFalse(screen.is_fullscreen) 

        display.toggle_fullscreen() 

        self.assertTrue(screen.is_fullscreen) 

        display.toggle_fullscreen() 

        self.assertFalse(screen.is_fullscreen) 

        expected = [call(True), call(False)] 

        window = mock_pyglet.window.Window.return_value 

        actual = window.set_fullscreen.call_args_list 

        self.assertEqual(expected, actual) 

 

    def test_update(self): 

        self.assertIsNone(display.update()) 

 

    def test_flip(self): 

        self.assertIsNone(display.flip()) 

 

    @patch("pygame2.display.pyglet") 

    def test_quit(self, mock_pyglet): 

        display.set_mode((0, 0)) 

        self.assertIsNone(display.quit()) 

        self.assertIsNone(display.PygameDisplay._instance) 

 

    @patch("pygame2.display.pyglet") 

    def test_get_surface(self, mock_pyglet): 

        self.assertIsNone(display.get_surface()) 

        screen = display.set_mode((0, 0)) 

        self.assertIs(screen, display.get_surface()) 

 

    @patch("pygame2.display.pyglet") 

    def test_get_init(self, mock_pyglet): 

        self.assertFalse(display.get_init()) 

        display.set_mode((0, 0)) 

        self.assertTrue(display.get_init())
